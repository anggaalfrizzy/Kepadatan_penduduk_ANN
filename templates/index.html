# Buka terminal, aktifkan venv, lalu install: pip install numpy pandas matplotlib seaborn tensorflow keras scikit-learn flask
import numpy as np import pandas as pd import matplotlib.pyplot as plt import seaborn as sns from sklearn.model_selection import train_test_split from sklearn.preprocessing import MinMaxScaler import tensorflow as tf from tensorflow.keras.models import Sequential from tensorflow.keras.layers import Dense
# Baca dataset dari file CSV OpenData Jabar df = pd.read_csv("Kepadatan_penduduk_jabar.csv") # Filter berdasarkan kabupaten tertentu df_kab = df[df["nama_kabupaten_kota"] == "KOTA BANDUNG"] df_kab = df_kab.sort_values("tahun") # Tampilkan 5 data pertama print(df_kab.head())
# Normalisasi dengan MinMaxScaler scaler_X = MinMaxScaler() scaler_y = MinMaxScaler() X = df_kab[["tahun"]].values y = df_kab[["kepadatan_penduduk"]].values X_scaled = scaler_X.fit_transform(X) y_scaled = scaler_y.fit_transform(y) # Split 80% train, 20% test X_train, X_test, y_train, y_test = train_test_split( X_scaled, y_scaled, test_size=0.2, random_state=42 )
# Membangun model ANN model = Sequential([ Dense(10, activation='relu', input_shape=(1,)), # Hidden 1 Dense(10, activation='relu'), # Hidden 2 Dense(1, activation='linear') # Output ]) # Kompilasi model model.compile(optimizer='adam', loss='mse', metrics=['mae']) # Training 200 epoch history = model.fit(X_train, y_train, epochs=200, validation_data=(X_test, y_test), verbose=1) # Evaluasi loss, mae = model.evaluate(X_test, y_test) print(f"MAE: {mae:.4f}")
# Prediksi tahun 2026 - 2030 tahun_pred = np.array([[2026],[2027],[2028],[2029],[2030]]) tahun_scaled = scaler_X.transform(tahun_pred) pred_scaled = model.predict(tahun_scaled) pred = scaler_y.inverse_transform(pred_scaled) for tahun, kepadatan in zip([2026,2027,2028,2029,2030], pred): print(f"Tahun {tahun}: {int(kepadatan[0])} jiwa/km²")
